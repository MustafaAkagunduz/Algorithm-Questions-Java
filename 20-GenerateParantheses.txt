class Solution {
    
    public List<String> generateParenthesis(int n) {
        List<String> res = new ArrayList<>();
        Stack<Character> stack = new Stack<>();
        backtrack(n, 0, 0, stack, res);
        return res;
    }

    private void backtrack(int n, int openCounter, int closedCounter, Stack<Character> stack, List<String> res) {
        if (openCounter == closedCounter && openCounter == n) {
            StringBuilder sb = new StringBuilder();
            for (char c : stack) {
                sb.append(c);
            }
            res.add(sb.toString());
            return;
        }

        if (openCounter < n) {
            stack.push('(');
            backtrack(n, openCounter + 1, closedCounter, stack, res);
            stack.pop();
        }
        if (closedCounter < openCounter) {
            stack.push(')');
            backtrack(n, openCounter, closedCounter + 1, stack, res);
            stack.pop();
        }
    }
}

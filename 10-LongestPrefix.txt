class Solution{
  if (strs == null || strs.length == 0) {
            return "";
        }

        // Diziyi sırala
        Arrays.sort(strs);

        // İlk ve son stringin karakterlerini karşılaştır
        String first = strs[0];
        String last = strs[strs.length - 1];
        int i = 0;

        while (i < first.length() && i < last.length() && first.charAt(i) == last.charAt(i)) {
            i++;
        }

        // Ortak önek
        return first.substring(0, i);
    }
}

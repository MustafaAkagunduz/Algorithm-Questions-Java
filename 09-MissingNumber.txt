class Solution{
	public int missingNumber(int [ ] nums){

	int sum = 0;
	int length = nums.length;

		for(int i = 0; i<nums.length; i++ ){

			sum += nums[ i ];

		}

		int olmasiGerekenTotal = (length*(length+1))/2;
		
		int result = olmasiGerekenTotal - sum 

		return result;

	}

}

public class MissingNumber {
    public static int findMissingNumber(int[] nums) {
        int n = nums.length;

        // Elemanları doğru indekslerine yerleştir
        for (int i = 0; i < n; ) {
            // Sayı doğru aralıkta mı? 
            // ve kendisiyle aynı indeks değilse
            if (nums[i] < n && nums[i] != nums[nums[i]]) {
                // Sayıyı doğru indeksine yerleştir
                int temp = nums[i];
                nums[i] = nums[temp];
                nums[temp] = temp; // Doğru yere koy
            } else {
                i++; // İndeks artır
            }
        }

        // Eksik sayıyı bul
        for (int i = 0; i < n; i++) {
            // İndeksle sayı eşleşmiyorsa eksik sayıyı bul
            if (nums[i] != i) {
                return i; // Eksik sayı
            }
        }

        // Eğer tüm sayılar doğru yerinde ise, eksik sayı n'dir
        return n;
    }

    public static void main(String[] args) {
        int[] nums = {3, 0, 1}; // Eksik sayı 2
        System.out.println("Eksik Sayı: " + findMissingNumber(nums));
    }
}
